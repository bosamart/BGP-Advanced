! ====================================================================
! ISP-A - upstream provider  (AS 65100)   Loopback0 100.100.100.100/32
! ---------------------------------------------------------------
! Simulates an upstream transit/peer. It originates:
!   8.8.8.0/24      <- shared "internet" prefix (ISP-B originates it too)
!   203.0.113.0/24  <- unique to ISP-A
! and learns 172.16.0.0/16 from our AS (65000) via R1.
! 'route-policy PASS' is the IOS-XR minimum to allow eBGP routes through.
! ====================================================================
hostname ISP-A
!
route-policy PASS
  pass
end-policy
!
! ===== Originated prefixes need a RIB route =====
router static
 address-family ipv4 unicast
  8.8.8.0/24 Null0
  203.0.113.0/24 Null0
 !
!
! ===== Interfaces =====
interface Loopback0
 ipv4 address 100.100.100.100 255.255.255.255
!
interface GigabitEthernet0/0/0/0          ! to R1
 ipv4 address 100.64.1.2 255.255.255.252
!
! ===== BGP =====
router bgp 65100
 bgp router-id 100.100.100.100
 address-family ipv4 unicast
  network 8.8.8.0/24
  network 203.0.113.0/24
 !
 neighbor 100.64.1.1
  remote-as 65000
  description eBGP-to-R1-AS65000
  address-family ipv4 unicast
   route-policy PASS in
   route-policy PASS out
  !
 !
!
end
